JAVA 发送代理，实现备忘

Posted on 2018-05-16 | In 技术 | 浏览 次

import com.baosight.bsms.;
import com.baosight.bsms.model.;
enum ReturnCode {
LoginFailed , SendMsgFailed , GetSendStatuFailed , LogOffFailed , NullParameter , Success;
}
interface Debug
{
public final boolean ENABLE = true;
}
public class JavaSendMsgProxy
{
String content;//要发送的短信内容
String numbers;//用逗号连接起来的要发送的号码
String connectURL; //登陆web service的url
String sendURL; //发送短信的URL
String account; //登陆的用户名
String password;//登陆时的密码
String identity; //登陆成功后URL返回的许可证
ReturnCode errorFlag; //错误描述
BsmsSendProxySoapBindingStub bsmsService ; //发送和接收等一系列操作需要的连接
BsmsAuthenticationSoapBindingStub authentication ;//连接成功后获得的授权对象
int priority ;//发送短信时的优先级
boolean needReturn; //是否等待短信发送回执
public String returnCode; // 用逗号连接起来的各个状态
public long messageId; //发送短信凭证
public int runFlag ; //程序执行错误码
public JavaSendMsgProxy()
{
showDebugInfo(“[constructor].constructor run ..”);
errorFlag=ReturnCode.Success;
this.connectURL=””;
this.sendURL=””;;
this.identity = null;
this.account = “GFZZGL”;
this.password = “2442goqx”;
this.returnCode=””;
}
public int logIn()
{
try
{
showDebugInfo(“[Login].Prepare to login …”);
authentication = new BsmsAuthenticationSoapBindingStub(new java.net.URL(connectURL),
null);
showDebugInfo(“[Login].Login action …”);
identity = authentication.login(account, password);
showDebugInfo(“[Login].after login identity >”+identity+”< …”);
}
catch(Exception ex)
{
errorFlag=ReturnCode.LoginFailed;
showDebugInfo(“[Login].login action exception occur”);
ex.printStackTrace();
}
return showError(errorFlag);
}
public int sendMessage(int priority,String content,String send_numbers)
{
if( content.equals(“”) || send_numbers.equals(“”) || content == null || send_numbers == null)
{
errorFlag = ReturnCode.NullParameter;
return showError(errorFlag);
}
try
{
showDebugInfo(“[send].receive content:”+content+”,numbers:”+send_numbers);
showDebugInfo(“[send].before sendProxy …”);
bsmsService = new BsmsSendProxySoapBindingStub(new java.net.URL(sendURL), null);
showDebugInfo(“[send].before messageId …”);
String numberArray[]=send_numbers.split(“,”);//将电话号码变为数组
messageId = bsmsService.sendMessage( numberArray ,content, priority, 0, identity);
showDebugInfo(“[send].send msg done ….”);
}
catch(Exception ex)
{
errorFlag=ReturnCode.SendMsgFailed;
showDebugInfo(“[send].send action exception occur”);
ex.printStackTrace();
}
return showError(errorFlag);
}
public int getSendStatus()
{
try
{
showDebugInfo(“[getSendStatus].before get status …”);
SentState[] states = bsmsService.getAllSentStatus(messageId, identity);
showDebugInfo(“[getSendStatus].after get status…”);
for (int i = 0; i < states.length; i++) {
returnCode = returnCode + states[i].getMobile().replaceAll(“ “,””) + “-“ + states[i].getState() + “,” ;
System.out.println (“接收手机号：” + states[i].getMobile());
System.out.println (“发送短信的接收时间：” + states[i].getReceiveTime());
System.out.println (“发送短信的接收状态：” + states[i].getState());
}
showDebugInfo(“[getSendStatus].returnCode”+returnCode+”<”);
}
catch(Exception ex)
{
errorFlag=ReturnCode.GetSendStatuFailed;
showDebugInfo(“[getSendStatus].Exception occur …”);
ex.printStackTrace();
}
return showError(errorFlag);
}
public int logOff()
{
try
{
showDebugInfo(“[logoff].before log off …”);
authentication.logoff(identity);
showDebugInfo(“[logoff].after log off …”);
}
catch(Exception ex)
{
errorFlag=ReturnCode.LogOffFailed;
showDebugInfo(“[logoff].Exception occur …”);
ex.printStackTrace();
}
return showError(errorFlag);
}
public int showError(ReturnCode e)
{
runFlag =0;
switch(e)
{
case LoginFailed :
showDebugInfo(“login action failed”);
runFlag = 1;
break;
case SendMsgFailed :
showDebugInfo(“send message action failed”);
runFlag = 2;
break;
case LogOffFailed :
showDebugInfo(“logoff action failed”);
runFlag = 3;
break;
case NullParameter :
showDebugInfo(“null parameter received”);
runFlag = 4;
break;
case GetSendStatuFailed :
showDebugInfo(“get send status failed”);
runFlag = 5;
break;
}
return runFlag;
}
public void showDebugInfo( String s)
{
if(Debug.ENABLE)
{
if( errorFlag == ReturnCode.Success )
System.out.println(“[DEBUG] : “+ s );
else
System.out.println(“[ERROR] : “+ s );
}
}
public static JavaSendMsgProxy SendMsg (int priority ,String content,String send_numbers)
{
System.out.println(“sendmsg receive priority:”+priority+”,content:”+content+”,numbers:”+send_numbers+”<”);
JavaSendMsgProxy sendMsgProxy=new JavaSendMsgProxy();
if ( sendMsgProxy.logIn() == 0 )
{
sendMsgProxy.sendMessage(priority,content,send_numbers) ;
sendMsgProxy.logOff() ;
}
return sendMsgProxy;
}
public static JavaSendMsgProxy SendMsgWithReturn (int priority ,String content,String send_numbers)
{
JavaSendMsgProxy sendMsgProxy=new JavaSendMsgProxy();
if ( sendMsgProxy.logIn() == 0 )
{
sendMsgProxy.sendMessage(priority,content,send_numbers) ;
sendMsgProxy.getSendStatus();
sendMsgProxy.logOff() ;
}
return sendMsgProxy;
}
public static void main(String [] args)
{
String con=”发送短信测试”;
String numbers=”18721903621,13889838492”;
//JavaSendMsgProxy j1 = SendMsg(1,con,numbers);
JavaSendMsgProxy j1 = SendMsgWithReturn(1,con,numbers);
}
}
